import requests
import bs4 as bs
ing = requests.get("https://viiiiiptips.blogspot.com/")
sauce = bs.BeautifulSoup(ing.content,'lxml')
tbl = sauce.find("table")
all_rows = tbl.find_all('tr')
_, _ = all_rows.pop(0), all_rows.pop(-1)
list_of_vals = []
def pick_out(row):
    for td in row.find_all('td'):
        val = td.text.strip()
        if len(val) != 0:
            list_of_vals.append(val)
r1, r2 = all_rows.pop(0), all_rows.pop(0)
pick_out(r1), pick_out(r2)

list_of_keys = ["time", "teams", "odds", "country", "3ways"]
meaty = {}
for k, v in zip(list_of_keys, list_of_vals):
    print(k, v)
